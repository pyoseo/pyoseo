title pyoseo GetStatus Operation

participant "Client" as client
participant "web server" as ws
participant "django" as dj
participant "oseoserver" as oseo
participant "GetStatus" as getstatus
participant "database" as db

client -> ws: OSEO GetStatus(order_id)
activate client
activate ws
ws -> dj: oseo_endpoint(request)
activate dj
dj -> oseo: process_request(request)
activate oseo
oseo -> oseo: authenticate_request()
oseo -> getstatus: process_request(request_data)
activate getstatus
getstatus -> db: get_order_status(order_id)
activate db
db -->> getstatus: order_status
deactivate db
getstatus-->>oseo: GetStatusResponse
deactivate getstatus
oseo -->> dj: GetstatusResponse
deactivate oseo
dj -->> ws: HttpResponse
deactivate dj
ws -->> client: OSEO GetStatusResponse
deactivate ws
deactivate client
